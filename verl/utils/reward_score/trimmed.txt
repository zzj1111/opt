CUDA_SRC = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>

#ifndef CHECK_CUDA
#define CHECK_CUDA(x) TORCH_CHECK(x.is_cuda(), #x " must be a CUDA tensor")
#endif

#ifndef CHECK_CONTIGUOUS
#define CHECK_CONTIGUOUS(x) TORCH_CHECK(x.is_contiguous(), #x " must be contiguous")
#endif

#define TILE_SIZE 32

/* -------------------------------------------------------------------------- */
/*                           Tiled MatMul Kernel                              */
/* -------------------------------------------------------------------------- */
__global__ void matmul_tiled_kernel(const float* __restrict__ A,
                                    const float* __restrict__ B,
                                    float* __restrict__ C,
                                    int M, int N, int K) {
    __shared__ float As[TILE_SIZE][TILE_SIZE];
    __shared__ float Bs[TILE_SIZE][TILE_SIZE];

    int row = blockIdx.y * TILE_SIZE + threadIdx.y;
    int col = blockIdx.x * TILE_SIZE + threadIdx.x;

    float acc = 0.f;

    // Loop over tiles of A/B in K dimension
    for (int t = 0; t < (K + TILE_SIZE - 1) / TILE_SIZE; ++t) {
        int tiledRow = t * TILE_SIZE + threadIdx.x;   // index in K for A
        int tiledCol = t * TILE_SIZE + threadIdx.y;   // index in K for B

        // Load A tile (M x K)
        if (row < M && tiledRow < K)
            As[threadIdx.y][threadIdx.x] = A[row * K + tiledRow];
        else
            As[threadIdx.y][threadIdx.x] = 0.f;

        // Load B tile (K x N)
        if (tiledCol < K && col < N)
            Bs[threadIdx.y][threadIdx.x] = B[tiledCol * N + col];
        else
            Bs[threadIdx.y][threadIdx.x] = 0.f;

        __syncthreads();

        // Compute partial product for this tile
        #pragma unroll
        for (int k = 0; k < TILE_SIZE; ++k)
            acc += As[threadIdx.y][k] * Bs[k][threadIdx.x];

        __syncthreads();
    }

    // Store the accumulated value
    if (row < M && col < N)
        C[row * N + col] = acc;
}

/* -------------------------------------------------------------------------- */
/*                      C++/CUDA Launcher Function                            */
/* -------------------------------------------------------------------------- */
torch::Tensor matmul_cuda(torch::Tensor A, torch::Tensor B) {
    CHECK_CUDA(A);
    CHECK_CUDA(B);
    CHECK_CONTIGUOUS(A);
    CHECK_CONTIGUOUS(B);
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "A and B must be 2D tensors");
    TORCH_CHECK(A.size(1) == B.size(0), "Inner dimensions must match");

    const int64_t M = A.size(0);
    const int64_t K = A.size(1);
    const int64_t N = B.size(1);

    auto C = torch::zeros({M, N}, A.options());

    dim3 block(TILE_SIZE, TILE_SIZE);
    dim3 grid((N + TILE_SIZE - 1) / TILE_SIZE,
              (M + TILE_SIZE - 1) / TILE_SIZE);

    matmul_tiled_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, N, K
    );

    // Optional: synchronize if you need correctness without subsequent kernel launches

    return C;
}
